**services/telegram_sync.py** — Create TelegramSync class:

```python
import logging
import tempfile
from pathlib import Path
from typing import Optional
import httpx

logger = logging.getLogger(__name__)

class TelegramSync:
    """Синхронизация локального приложения с сервером Telegram-бота."""

    def __init__(self, server_url: str, chat_id: int):
        self.base = server_url.rstrip("/")
        self.chat_id = chat_id
        self._client = httpx.Client(timeout=30)

    def is_configured(self) -> bool:
        return bool(self.base) and self.chat_id > 0

    def bind(self, code: str) -> Optional[int]:
        """Send binding code to server, return chat_id on success."""
        try:
            r = self._client.post(f"{self.base}/api/bind", params={"code": code})
            r.raise_for_status()
            data = r.json()
            return data.get("chat_id")
        except httpx.HTTPError as e:
            logger.warning("Binding failed: %s", e)
            return None

    def fetch_queue(self, dest_dir: Path) -> list[Path]:
        """Fetch all queued files from server, save to dest_dir, return local paths."""
        if not self.is_configured():
            return []
        try:
            r = self._client.get(f"{self.base}/api/queue/{self.chat_id}")
            r.raise_for_status()
            files_data = r.json().get("files", [])
        except httpx.HTTPError as e:
            logger.warning("Failed to fetch queue: %s", e)
            return []

        dest_dir.mkdir(parents=True, exist_ok=True)
        paths = []
        for item in files_data:
            try:
                fr = self._client.get(f"{self.base}/api/files/{item['id']}")
                fr.raise_for_status()
                dest = dest_dir / item["filename"]
                dest.write_bytes(fr.content)
                # Confirm fetched
                self._client.delete(f"{self.base}/api/queue/{item['id']}")
                paths.append(dest)
            except httpx.HTTPError as e:
                logger.warning("Failed to fetch file %s: %s", item.get("filename"), e)
        return paths

    def push_deadlines(self, alerts: list[dict]) -> bool:
        """Push minimal deadline metadata to server for cron notifications."""
        if not self.is_configured():
            return False
        try:
            r = self._client.post(
                f"{self.base}/api/deadlines/{self.chat_id}",
                json={"alerts": alerts},
            )
            r.raise_for_status()
            return True
        except httpx.HTTPError as e:
            logger.warning("Failed to push deadlines: %s", e)
            return False

    def notify_processed(self, chat_id: int, summary: str) -> bool:
        """Ask bot server to send processing result card to user."""
        if not self.is_configured():
            return False
        try:
            r = self._client.post(
                f"{self.base}/api/notify",
                json={"chat_id": chat_id, "text": summary},
            )
            r.raise_for_status()
            return True
        except httpx.HTTPError as e:
            logger.warning("Failed to notify: %s", e)
            return False
```

# --- Fetch & process Telegram queue (INTG-01) ---
if not st.session_state.get("tg_queue_fetched") and config.telegram_server_url and config.telegram_chat_id > 0:
    from services.telegram_sync import TelegramSync
    from services.pipeline_service import PipelineService

        _results = _pipeline.process_archive(_tg_dest)
        # Send processing summary back to Telegram
        _success = sum(1 for r in _results if r.success)
        _failed = len(_results) - _success
        _summary_parts = [f"Обработано {_success} из {len(_results)} файлов."]
        for r in _results:
            if r.success and r.metadata:
                m = r.metadata
                _summary_parts.append(
                    f"- {m.doc_type}: {m.counterparty or 'н/д'}, "
                    f"{m.total_amount or 'н/д'}, до {m.date_end or 'н/д'}"
                )
        _sync.notify_processed(config.telegram_chat_id, "\n".join(_summary_parts))
        st.success(f"Telegram: обработано {_success} файлов" + (f", ошибок: {_failed}" if _failed else ""))
    st.session_state["tg_queue_fetched"] = True
