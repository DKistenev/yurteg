Вкладка «Платёжный календарь» — реализация:
```python
# В блоке вкладки «Платёжный календарь»:
st.header("Платёжный календарь")

if db is None:
    st.info("Сначала обработайте архив документов")
else:
    events = get_calendar_events(db)
    if not events:
        st.info("Платёжные данные отсутствуют. Загрузите договоры с указанными суммами и сроками.")
    else:
        # Показать сводку
        expense_total = sum(e["extendedProps"]["amount"] for e in events if e["extendedProps"]["direction"] == "expense")
        income_total = sum(e["extendedProps"]["amount"] for e in events if e["extendedProps"]["direction"] == "income")
        col_e, col_i = st.columns(2)
        with col_e:
            st.metric("Расходы (всего)", f"{expense_total:,.0f} ₽".replace(",", " "))
        with col_i:
            st.metric("Доходы (всего)", f"{income_total:,.0f} ₽".replace(",", " "))

        # Календарь
        calendar_options = {
            "initialView": "dayGridMonth",
            "headerToolbar": {
                "left": "prev,next today",
                "center": "title",
                "right": "dayGridMonth,timeGridWeek",
            },
            "locale": "ru",
            "height": 600,
        }
        clicked = st_calendar(events=events, options=calendar_options, key="payment_calendar")

        # Обработка клика по событию
        if clicked and clicked.get("eventClick"):
            event_data = clicked["eventClick"]["event"]
            props = event_data.get("extendedProps", {})
            direction_label = "Расход" if props.get("direction") == "expense" else "Доход"
            amount_str = f"{props.get('amount', 0):,.0f} ₽".replace(",", " ")
            with st.expander("Детали платежа", expanded=True):
                st.markdown(f"**{event_data.get('title', '—')}**")
                st.markdown(f"**Тип:** {direction_label}")
                st.markdown(f"**Сумма:** {amount_str}")
                st.markdown(f"**Контрагент:** {props.get('counterparty', '—')}")
                st.markdown(f"**Тип договора:** {props.get('contract_type', '—')}")
                st.markdown(f"**ID договора:** {props.get('contract_id', '—')}")
```
