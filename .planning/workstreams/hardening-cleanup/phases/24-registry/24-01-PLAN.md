**Структура render_split_panel():**

```python
def render_split_panel(container, doc, on_close, on_open_full):
    container.clear()
    if doc is None:
        container.set_visibility(False)
        return
    container.set_visibility(True)
    with container:
        # TOP: контрагент + ✕
        with ui.row().classes("px-5 py-4 border-b border-slate-100 justify-between items-start w-full"):
            with ui.column().classes("gap-1"):
                ui.label(doc.get("counterparty", "—")).style("font-size:15px; font-weight:600; color:#1e293b;")
                # Тег-бейдж типа
                ui.label(doc.get("contract_type", "")).classes(PANEL_TYPE_TAG)
            ui.button("✕", on_click=on_close).props("flat dense").style("color:#94a3b8; font-size:18px; padding:0; min-width:auto;")

        # Секция ДОКУМЕНТ
        with ui.column().classes("px-5 py-3 border-b border-slate-50 gap-1 w-full"):
            ui.label("Документ").classes(PANEL_SEC_TITLE)
            _field_linear("Предмет", doc.get("subject", "—"))
            # Status badge
            status = doc.get("computed_status", "unknown")
            badge_cls, badge_label = _STATUS_STYLE.get(status, _STATUS_STYLE["unknown"])
            ui.label(badge_label).classes(f"mt-1 px-2.5 py-0.5 text-[11px] font-medium rounded-full {badge_cls}")

        # Секция СРОКИ
        with ui.column().classes("px-5 py-3 border-b border-slate-50 w-full"):
            ui.label("Сроки").classes(PANEL_SEC_TITLE)
            with ui.row().classes("grid grid-cols-2 gap-2 w-full"):
                _field_linear("Начало", doc.get("date_start", "—"))
                _field_linear("Окончание", doc.get("date_end", "—"))

        # Секция ФИНАНСЫ
        with ui.column().classes("px-5 py-3 w-full"):
            ui.label("Финансы").classes(PANEL_SEC_TITLE)
            amount_str = _format_amount(doc.get("amount"))
            ui.label(amount_str).style("font-size:20px; font-weight:700; color:#1e293b; letter-spacing:-0.02em;")

        # Кнопка "Открыть карточку →"
        ui.button("Открыть карточку →", on_click=on_open_full).style(
            "margin:12px 16px 16px; padding:8px; border:1px solid #e2e8f0; border-radius:8px; "
            "background:white; color:#475569; font-size:12px; font-weight:500; width:calc(100% - 32px); "
            "text-align:center; font-family:inherit;"
        ).props("flat no-caps")
```
