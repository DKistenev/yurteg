    1. Add version grouping to `app/components/registry_table.py`:

       ```python
       def build_version_rows(base_rows: list[dict], db) -> list[dict]:
           """Build flat list with version children. Per D-15, D-16, D-17.
           Children are hidden by default (_hidden=True). Expand toggle handled in UI.
           Uses lazy approach — only marks parents, children loaded on expand click.
           """
           from services.version_service import get_version_group
           # Bulk check: which contract_ids have versions
           ids = [r["id"] for r in base_rows]
           if not ids:
               return base_rows
           placeholders = ",".join("?" * len(ids))
           version_counts = {}
           try:
               cursor = db.conn.execute(
                   f"SELECT contract_id, contract_group_id FROM document_versions WHERE contract_id IN ({placeholders})",
                   ids
               )
               groups = {}
               for row in cursor:
                   gid = row["contract_group_id"]
                   cid = row["contract_id"]
                   groups.setdefault(gid, []).append(cid)
               # A parent has children if its group has >1 member
               for gid, members in groups.items():
                   if len(members) > 1:
                       for cid in members:
                           version_counts[cid] = len(members)
           except Exception:
               pass  # document_versions table may not exist in older DBs

           result = []
           for row in base_rows:
               row["has_children"] = row["id"] in version_counts and version_counts[row["id"]] > 1
               row["is_child"] = False
               row["indent"] = 0
               result.append(row)
           return result
       ```
