```python
"""Ollama провайдер — локальная LLM через llama-server.

Использует OpenAI-совместимый endpoint llama-server на localhost.
Post-processing ответов через modules/postprocessor.sanitize_metadata.
"""
import json
import logging

from openai import OpenAI

from config import Config
from providers.base import LLMProvider

logger = logging.getLogger(__name__)


class OllamaProvider(LLMProvider):
    """Провайдер для llama-server — локальная QWEN 1.5B через OpenAI API."""

    name = "ollama"

    def __init__(self, config: Config, base_url: str = "http://localhost:8080/v1") -> None:
        self._config = config
        self._client = OpenAI(
            base_url=base_url,
            api_key="not-needed",  # llama-server не требует ключ
        )

    def complete(self, messages: list[dict], **kwargs) -> str:
        response = self._client.chat.completions.create(
            model="local",  # llama-server игнорирует model name
            temperature=0.05,  # из Modelfile
            max_tokens=512,   # num_predict из Modelfile
            messages=messages,
        )
        content = response.choices[0].message.content
        if not content:
            raise RuntimeError("llama-server вернул пустой ответ")
        return content

    def verify_key(self) -> bool:
        """Проверяет доступность llama-server через health endpoint."""
        try:
            self._client.models.list()
            return True
        except Exception:
            return False
```

**Ключевые решения:**
- `base_url` передаётся при создании (из LlamaServerManager.base_url) — не хардкодить в конструкторе, но дефолт "http://localhost:8080/v1"
- `api_key="not-needed"` — openai SDK требует непустой ключ, llama-server его игнорирует
- `model="local"` — llama-server загружает модель при старте, имя игнорируется
- `temperature=0.05`, `max_tokens=512` — из Modelfile параметров
- Post-processing НЕ вызывается внутри провайдера — он возвращает сырой текст, sanitize_metadata будет применяться в ai_extractor.py (Phase 5, PROC-01)

```python
from services.llama_server import LlamaServerManager
from modules.postprocessor import get_grammar_path

# ── Автозапуск llama-server для локальной LLM ──────
@st.cache_resource
def _get_llama_manager() -> LlamaServerManager | None:
    """Запускает llama-server один раз, переживает Streamlit reruns."""
    config = Config()
    if config.active_provider != "ollama":
        return None

    manager = LlamaServerManager(port=config.llama_server_port)

    # Скачивание модели
    try:
        with st.spinner("Скачивание модели (~940 МБ)..."):
            manager.ensure_model()
    except Exception as e:
        st.warning(f"Не удалось скачать модель: {e}. Переключение на облачный провайдер.")
        return None

    # Скачивание llama-server
    try:
        with st.spinner("Скачивание llama-server..."):
            manager.ensure_server_binary()
    except Exception as e:
        st.warning(f"Не удалось скачать llama-server: {e}. Переключение на облачный провайдер.")
        return None

    # Запуск сервера
    if not manager.is_running():
        try:
            grammar = get_grammar_path()
            manager.start(grammar_path=grammar)
        except Exception as e:
            st.warning(f"llama-server не запустился: {e}. Переключение на облачный провайдер.")
            return None

    return manager

_llama = _get_llama_manager()
if _llama is None and Config().active_provider == "ollama":
    st.warning("⚠ Локальная модель недоступна. Используется облачный провайдер.")
    # Переопределить провайдер на fallback для текущей сессии
    # Не менять config.py — только runtime
```

**Важно:** Использовать `@st.cache_resource` чтобы сервер не перезапускался при каждом Streamlit rerun (blocker из STATE.md). `cache_resource` сохраняет объект между reruns.
