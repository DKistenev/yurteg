# ── AI Review section (per D-10) ──
with ui.expansion('Ревью', icon='rate_review').classes('w-full border rounded-lg'):
    review_container = ui.column().classes('w-full gap-2')

    async def _run_review():
        review_container.clear()
        with review_container:
            spinner = ui.spinner('dots').classes('text-blue-500')

        db = _client_manager.get_db(state.current_client)
        # Per D-11: auto-match template first
        template = await run.io_bound(
            match_template, db, contract.get('subject', ''), contract.get('contract_type')
        )
        if template is None:
            # Check if any templates exist at all
            templates = await run.io_bound(list_templates, db)
            if not templates:
                # Per D-14: no templates message with link
                review_container.clear()
                with review_container:
                    with ui.row().classes('items-center gap-2 text-gray-500 text-sm'):
                        ui.label('Нет шаблонов.')
                        ui.link('Добавьте в разделе Шаблоны', '/templates').classes('text-blue-600 underline')
                return
            else:
                # Show dropdown for manual selection (per D-11 fallback)
                review_container.clear()
                with review_container:
                    template_options = {t.id: f"{t.name} ({t.contract_type})" for t in templates}
                    selected_template = ui.select(
                        template_options,
                        label='Выберите шаблон',
                    ).classes('w-full max-w-sm')

                    async def _review_with_selected():
                        sel_id = selected_template.value
                        if sel_id is None:
                            return
                        sel_tmpl = next((t for t in templates if t.id == sel_id), None)
                        if sel_tmpl:
                            await _do_review(sel_tmpl.content_text)

                    ui.button('Проверить', on_click=_review_with_selected).props('flat no-caps').classes('text-blue-600')
                return

        # Template found — run review directly
        await _do_review(template.content_text)

    async def _do_review(template_text: str):
        review_container.clear()
        with review_container:
            spinner = ui.spinner('dots').classes('text-blue-500')
        # Per D-12: async via run.io_bound
        deviations = await run.io_bound(
            review_against_template, template_text, contract.get('subject', '')
        )
        _render_deviations(review_container, deviations)

    ui.button('Проверить по шаблону', on_click=_run_review).props('flat no-caps').classes('text-blue-600')
```

**Helper function `_render_deviations` (per D-13, from RESEARCH code example):**

def _render_deviations(container, deviations: list[dict]) -> None:
    TYPE_LABEL = {"added": "Добавлено", "removed": "Удалено", "changed": "Изменено"}
    container.clear()
    with container:
        if not deviations:
            ui.label("Отступлений не найдено").classes('text-green-600 text-sm')
            return
        for d in deviations:
            # Per Pitfall 4: inline style for hex color, NOT dynamic Tailwind class
            ui.html(
                f'<div style="border-left: 3px solid {d["color"]}; padding: 8px 12px; '
                f'background: {d["color"]}22; border-radius: 6px; margin-bottom: 8px;">'
                f'<div style="font-size: 11px; color: #9ca3af; margin-bottom: 4px;">{TYPE_LABEL.get(d["type"], d["type"])}</div>'
                + (f'<div style="font-size: 12px; color: #6b7280; text-decoration: line-through;">{d.get("template_text") or ""}</div>' if d.get("template_text") else '')
                + (f'<div style="font-size: 14px; color: #111827;">{d.get("document_text") or ""}</div>' if d.get("document_text") else '')
                + '</div>'
            )

def _render_diff_table(container, diffs: list[dict]) -> None:
    changed = [d for d in diffs if d['changed']]
    if not changed:
        with container:
            ui.label('Изменений не найдено').classes('text-green-600 text-sm')
        return
    with container:
        columns = [
            {'name': 'field', 'label': 'Поле', 'field': 'field', 'align': 'left'},
            {'name': 'old', 'label': 'Было', 'field': 'old', 'align': 'left'},
            {'name': 'new', 'label': 'Стало', 'field': 'new', 'align': 'left'},
        ]
        table = ui.table(columns=columns, rows=changed).classes('w-full text-sm')
        table.add_slot('body-cell-old', '<q-td :props="props"><span class="text-red-600 line-through">{{ props.value }}</span></q-td>')
        table.add_slot('body-cell-new', '<q-td :props="props"><span class="text-green-700">{{ props.value }}</span></q-td>')

# ── Redline download route (Phase 9, D-18) ─────────────────────────────────
from fastapi.responses import Response as FastAPIResponse

@app.get('/download/redline/{contract_id}/{other_id}')
async def download_redline(contract_id: int, other_id: int):
    """Download redline .docx comparing two contract versions."""
    from services.version_service import generate_redline_docx
    from services.client_manager import ClientManager

    cm = ClientManager()
    # Use default client — redline is requested from document card context
    db = cm.get_db("Основной реестр")
    c1 = await run.io_bound(db.get_contract_by_id, contract_id)
    c2 = await run.io_bound(db.get_contract_by_id, other_id)
    if c1 is None or c2 is None:
        return FastAPIResponse(content="Документ не найден", status_code=404)
    text_old = c1.get('subject', '')
    text_new = c2.get('subject', '')
    title = f"Redline: {c1.get('contract_type', '')} vs {c2.get('contract_type', '')}"
    docx_bytes = await run.io_bound(generate_redline_docx, text_old, text_new, title)
    return FastAPIResponse(
        content=docx_bytes,
        media_type='application/vnd.openxmlformats-officedocument.wordprocessingml.document',
        headers={'Content-Disposition': f'attachment; filename="redline_{contract_id}_vs_{other_id}.docx"'},
    )
```

NOTE: The redline download route uses `subject` as proxy text because full document text is not stored in DB (per STATE.md Phase 2 decision). The `app` object is the NiceGUI `app` already imported at the top.
